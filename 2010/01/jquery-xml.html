Javascript (讀取 XML 檔案):
$(document).ready(function(){
 var xml = $.ajax({
  url: 'google_news.xml',
        async: false
 }).responseXML;
 var $body = $("body");
 $(xml).find("item").each(function(i){
  var $item = $(this);
  $body.append($("<p>").append($("<a>").attr({
   href: $item.children("link").text()
  }).text(i + 1 + ". " + $item.children("title").text())))
 });
});
